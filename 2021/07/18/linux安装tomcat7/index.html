Linux安装tomcat7

发表于 2021-07-18

1.安装Tomcat

使用xftp上传apache-tomcat-7.0.109.tar.gz到 tomcat7
[root@localhost ~]# cd /usr/tomcat7
[root@localhost tomcat7]# ls
apache-tomcat-7.0.109.tar.gz
[root@localhost tomcat7]# tar zxvf apache-tomcat-7.0.109.tar.gz

2.环境变量

[root@localhost tomcat7]# vim /etc/profile

export CATALINA_HOME=/usr/tomcat7
export PATH=${CATALINA_HOME}/bin:$PATH

[root@localhost tomcat7]# source /etc/profile

3.配置catalina.sh文件
vim /usr/tomcat7/apache-tomcat-7.0.109/bin/catalina.sh

在头部添加

JAVA_HOME=/usr/jdk1.8.0_221
JRE_HOME=${JAVA_HOME}/jre

4.启动tomcat服务

[root@localhost ~]# cd /usr/tomcat7/apache-tomcat-7.0.109/bin
[root@localhost bin]# ./startup.sh

5.开放8080端口

#放行8080端口
[root@localhost bin]# firewall-cmd –zone=public –add-port=8080/tcp –permanent
success
#重启防火墙
[root@localhost bin]# firewall-cmd –reload
success
6.增加 manager-gui 图形化管理界面的访问权限

[root@localhost bin]# vim /usr/tomcat7/apache-tomcat-7.0.109/conf/tomcat-users.xml

#添加如下代码

#修改webapps/manager/META-INF目录下的context.xml,在allow行的末尾加上|\d+.\d+.\d+.\d+表示允许所有主机访问。
[root@localhost /]# cd /usr/tomcat7/apache-tomcat-7.0.109/webapps/manager/META-INF
[root@localhost META-INF]# vim context.xml

<Context antiResourceLocking="false" privileged="true" >
<Valve className="org.apache.catalina.valves.RemoteAddrValve"
     allow="127\.\d+\.\d+\.\d+|::1|0:0:0:0:0:0:0:1|\d+.\d+.\d+.\d+" />
<Manager sessionAttributeValueClassNameFilter="java\.lang\.(?:Boolean|Integer|Long|Number|String)|org\.apache\.catalina\.filters\.CsrfPreventionFilter\$LruCache(?:\$1)?|java\.util\.(?:Linked)?HashMap"/>
</Context>

重启tomcat生效

[root@localhost META-INF]# cd /usr/tomcat7/apache-tomcat-7.0.109/bin/

[root@localhost bin]# ./startup.sh

7.配置tomcat开机自启

#在/usr/lib/systemd/system目录下增加tomcat.service文件，内容如下：

[root@localhost bin]# cd /usr/lib/systemd/system

[root@localhost system]# vim tomcat.service

[Unit]

Description=Tomcat

After=syslog.target network.target remote-fs.target nss-lookup.target


[Service]

Type=oneshot

ExecStart=/usr/tomcat7/apache-tomcat-7.0.109/bin/startup.sh

ExecStop=/usr/tomcat7/apache-tomcat7.0.109/bin/shutdown.sh

ExecReload=/bin/kill -s HUP $MAINPID

RemainAfterExit=yes


[Install]

WantedBy=multi-user.target

退出并保存，执行systemctl enable tomcat.service使tomcat随开机启动

[root@localhost system]# systemctl enable tomcat.service

[root@localhost system]# systemctl start tomcat.service #启动tomcat

[root@localhost system]# systemctl stop tomcat.service #关闭tomcat

[root@localhost system]# systemctl restart tomcat.service # 重启tomcat
